defineVariable(float, COST_SHIP_FACTOR);
defineVariable(float, COST_AIR_FACTOR);
defineVariable(int, CAN_IMPORT_SERVICES);

defineFunction(InitSettings, void)
{
	//you can modify the following values to enable or disable features or change the balance, making the game harder or easier
	//please keep at least one digit after the decimal point if present
	
	INFORMATION_SHOWN = 0; //2 - full information (3 lines), 1 - concise information (1 line), 0 - turned off
	
	START_YEAR = 0; //starting year of the game, if left to 0 the script will try to determine it based on your import history
	INFLATION_RUB = 1.09; //rubles inflation rate
	INFLATION_USD = 1.027; //dollars inflation rate
	INTERNAL_DEMAND_FACTOR = 0.02; //how many workers should be employed in offices to serve 1 citizen (setting it to 0.0 or less will disable the feature)
	GRACE_CITIZENS = 7500; //the demand for services is decreased for your first GRACE_CITIZENS non-dormant citizens
	GRACE_FACTOR = 0.5; //the decrease in demand for your first GRACE_CITIZENS non-dormant citizens
	NUMBER_OF_PRICE_DECREASES = 0; //how many times have you used cheats to divide prices by 100, every time you use it in your save you should increase it by 1 and restart the script, otherwise the cost of services will not be divided by 100
	
	CAN_IMPORT_SERVICES = 1; //1 or more allows you to import services from abroad if you need them, with 0 or less you have to generate the necessary services yourself
	
	//base prices of 1 unit of services
	PRICE_BUY_RUB = 6.0;
	PRICE_BUY_USD = 8.0;
	PRICE_SELL_RUB = 5.0;
	PRICE_SELL_USD = 6.5;
	
	BASE_FOREIGN_SUPPLY_DEMAND = 200.0; //the foreign supply and demand works as inertia for calculating the current market price of services
	FOREIGN_SUPPLY_DEMAND_FACTOR = 0.03; //how much your population increases the foreign supply and demand
	
	PRIMARY_ED_FACTOR = 1.0; //services generated by a worker with primary education
	UNIVERSITY_ED_FACTOR = 1.75; //services generated by a worker with university education
	HQ_FACTOR = 1.5; //services generated by a university educated worker in a party HQ
	
	//the following constants determine the negative influence of not being able to meet the need for services
	//setting to 0 will disable the corresponding effect, negative values will result in a positive influence
	UNHAPPINESS_FACTOR = 0.005; //-0.5 percent points per day
	LOYALTY_FACTOR = 0.0025; //-0.25 percent points per day
	CRIME_FACTOR = 0.005; //+0.5 percent points per day
	
	FIRE_PROB_DENOMINATOR = 100; //the additional probability per tick (~10 times a day) of fire is 1 over this, setting it to 0 or less will disable the feature
	
	NO_POWER_FACTOR = 0.25; //how much your generation of services is scaled when the office building lacks electricity 
	NO_WATER_FACTOR = 0.5; //how much your generation of services is scaled when the office building lacks water 

	// Scale factor for all vehicle-related costs. 0.0 disables the feature. 0.5 makes them use half as much supply.
	VEHICLE_DIFFICULTY = 1.0;

	// Idle factor. Vehicles that are in their depot or office are not in use and will have their cost multiplied by this
	VEHICLE_IDLE_COST = 0.01;

	//base administrative cost (in units of services) needed by each vehicle regardless of its resource capacity (currently unused due to a game bug)
	BASE_COST_ROAD = 1.0;
	BASE_COST_RAIL = 2.0;
	BASE_COST_SHIP = 10.0;
	BASE_COST_AIR = 4.0;
	
	//base administrative cost for seasonal vehicles
	BASE_COST_SEEDING = 1.25;
	BASE_COST_HARVESTING = 1.25;
	BASE_COST_SNOWPLOWING = 1.5;
	
	//base administrative cost for personal cars
	BASE_COST_PERSONAL = 0.02;
	
	//administrative cost needed by each vehicle per each 1 t of resources or 10 passengers
	COST_ROAD_FACTOR = 0.02;
	COST_RAIL_FACTOR = 0.002;
	COST_SHIP_FACTOR = 0.0002;
	COST_AIR_FACTOR = 0.05;
	
	returnVoid();
}
